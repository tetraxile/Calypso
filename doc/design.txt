                switch (command->type) {
                case Type::CONTROLLER: {
                    ControllerCommand* cmd = (ControllerCommand*)command;
                    packet.controller = *cmd;
                    break;
                }
                case Type::MOTION: {
                    MotionCommand* cmd = (MotionCommand*)command;
                    packet.motion = *cmd;
                    break;
                }
                case ...: {

                }
                }
            }
        }
